```csharp
public class VirtualizingWrapPanel : VirtualizingPanel, IScrollInfo
{
    protected override Size MeasureOverride(Size availableSize)
    {
        // 只测量可见项
        UpdateScrollInfo(availableSize);
        
        int firstVisible = GetFirstVisibleIndex();
        int lastVisible = GetLastVisibleIndex();
        
        var generator = ItemContainerGenerator;
        var startPos = generator.GeneratorPositionFromIndex(firstVisible);
        
        using (generator.StartAt(startPos, GeneratorDirection.Forward, true))
        {
            for (int i = firstVisible; i <= lastVisible; i++)
            {
                UIElement child = generator.GenerateNext(out bool isNewlyRealized) as UIElement;
                if (isNewlyRealized)
                {
                    AddInternalChild(child);
                }
                child.Measure(GetItemSize());
            }
        }
        
        return availableSize;
    }
    
    private int GetFirstVisibleIndex()
    {
        return (int)(_offset.Y / _itemHeight);
    }
    
    private int GetLastVisibleIndex()
    {
        return Math.Min(
            (int)((_offset.Y + _viewport.Height) / _itemHeight) + 1,
            ItemCount - 1
        );
    }
}
```

### 实现：高性能图表控件

```csharp
public class HighPerformanceChart : FrameworkElement
{
    private DrawingVisual _visual;
    private Point[] _dataPoints;
    
    public void SetData(double[] values)
    {
        _dataPoints = ConvertToPoints(values);
        InvalidateVisual();
    }
    
    protected override void OnRender(DrawingContext dc)
    {
        if (_dataPoints == null || _dataPoints.Length == 0)
            return;
        
        // 绘制背景
        dc.DrawRectangle(Brushes.White, null, new Rect(RenderSize));
        
        // 绘制网格（减少数量）
        DrawGrid(dc, 10); // 只画10条线
        
        // 绘制数据线（使用StreamGeometry优化）
        var geometry = new StreamGeometry();
        using (var ctx = geometry.Open())
        {
            ctx.BeginFigure(_dataPoints[0], false, false);
            
            // 数据采样（如果点太多）
            var sampledPoints = SamplePoints(_dataPoints, 10000);
            for (int i = 1; i < sampledPoints.Length; i++)
            {
                ctx.LineTo(sampledPoints[i], true, false);
            }
        }
        geometry.Freeze(); // 冻结以提高性能
        
        dc.DrawGeometry(null, new Pen(Brushes.Blue, 1), geometry);
    }
    
    private Point[] SamplePoints(Point[] points, int maxCount)
    {
        if (points.Length <= maxCount)
            return points;
        
        // 抽样算法（保留峰值）
        var step = (double)points.Length / maxCount;
        var result = new List<Point>();
        
        for (int i = 0; i < maxCount; i++)
        {
            int start = (int)(i * step);
            int end = (int)((i + 1) * step);
            
            // 保留该区间的最大和最小值
            var segment = points.Skip(start).Take(end - start);
            result.Add(segment.OrderBy(p => p.Y).First()); // min
            result.Add(segment.OrderBy(p => p.Y).Last());  // max
        }
        
        return result.ToArray();
    }
    
    protected override int VisualChildrenCount => 1;
    protected override Visual GetVisualChild(int index) => _visual;
}
```

### 增量渲染

```csharp
public class IncrementalChart : FrameworkElement
{
    private Queue<Point> _pendingPoints = new Queue<Point>();
    private DispatcherTimer _timer;
    
    public void AddData(Point[] newPoints)
    {
        foreach (var point in newPoints)
        {
            _pendingPoints.Enqueue(point);
        }
        
        StartIncrementalRender();
    }
    
    private void StartIncrementalRender()
    {
        if (_timer == null)
        {
            _timer = new DispatcherTimer(DispatcherPriority.Render);
            _timer.Interval = TimeSpan.FromMilliseconds(16); // 60fps
            _timer.Tick += (s, e) =>
            {
                // 每帧渲染1000个点
                int count = Math.Min(1000, _pendingPoints.Count);
                for (int i = 0; i < count; i++)
                {
                    AddPointToVisual(_pendingPoints.Dequeue());
                }
                
                InvalidateVisual();
                
                if (_pendingPoints.Count == 0)
                {
                    _timer.Stop();
                }
            };
        }
        
        _timer.Start();
    }
}
```

#### 使用WriteableBitmap

```csharp
public class PixelCanvas : Image
{
    private WriteableBitmap _bitmap;
    
    public void Initialize(int width, int height)
    {
        _bitmap = new WriteableBitmap(
            width, height, 
            96, 96, 
            PixelFormats.Bgra32, 
            null
        );
        Source = _bitmap;
    }
    
    public void DrawPixel(int x, int y, Color color)
    {
        _bitmap.Lock();
        try
        {
            unsafe
            {
                int pBackBuffer = (int)_bitmap.BackBuffer;
                pBackBuffer += y * _bitmap.BackBufferStride;
                pBackBuffer += x * 4;
                
                int colorData = color.A << 24;
                colorData |= color.R << 16;
                colorData |= color.G << 8;
                colorData |= color.B;
                
                *((int*)pBackBuffer) = colorData;
            }
            
            _bitmap.AddDirtyRect(new Int32Rect(x, y, 1, 1));
        }
        finally
        {
            _bitmap.Unlock();
        }
    }
    
    // 批量绘制（更高效）
    public void DrawPixels(Point[] points, Color color)
    {
        _bitmap.Lock();
        try
        {
            unsafe
            {
                int colorData = (color.A << 24) | (color.R << 16) | (color.G << 8) | color.B;
                
                foreach (var p in points)
                {
                    int x = (int)p.X;
                    int y = (int)p.Y;
                    
                    int* pBackBuffer = (int*)_bitmap.BackBuffer;
                    pBackBuffer += y * _bitmap.BackBufferStride / 4;
                    pBackBuffer += x;
                    
                    *pBackBuffer = colorData;
                }
            }
            
            _bitmap.AddDirtyRect(new Int32Rect(0, 0, _bitmap.PixelWidth, _bitmap.PixelHeight));
        }
        finally
        {
            _bitmap.Unlock();
        }
    }
}
```

```csharp
// ✅ 使用弱引用事件
public class WeakEventManager<TEventArgs>
{
    private List<WeakReference<EventHandler<TEventArgs>>> _handlers = new();
    
    public void AddHandler(EventHandler<TEventArgs> handler)
    {
        _handlers.Add(new WeakReference<EventHandler<TEventArgs>>(handler));
    }
    
    public void RemoveHandler(EventHandler<TEventArgs> handler)
    {
        _handlers.RemoveAll(wr =>
        {
            if (wr.TryGetTarget(out var target))
                return target == handler;
            return true; // 移除已释放的引用
        });
    }
    
    public void Raise(object sender, TEventArgs args)
    {
        _handlers = _handlers.Where(wr =>
        {
            if (wr.TryGetTarget(out var handler))
            {
                handler(sender, args);
                return true;
            }
            return false; // 移除已释放的引用
        }).ToList();
    }
}
```

```csharp
// ❌ 无限增长的缓存
public class ImageCache
{
    private Dictionary<string, BitmapImage> _cache = new();
    
    public BitmapImage GetImage(string path)
    {
        if (!_cache.ContainsKey(path))
        {
            _cache[path] = LoadImage(path);
        }
        return _cache[path];
    }
}

// ✅ LRU缓存
public class LRUImageCache
{
    private readonly int _capacity;
    private readonly Dictionary<string, LinkedListNode<CacheItem>> _cache;
    private readonly LinkedList<CacheItem> _lruList;
    
    public LRUImageCache(int capacity)
    {
        _capacity = capacity;
        _cache = new Dictionary<string, LinkedListNode<CacheItem>>(capacity);
        _lruList = new LinkedList<CacheItem>();
    }
    
    public BitmapImage GetImage(string path)
    {
        if (_cache.TryGetValue(path, out var node))
        {
            // 移到最前面（最近使用）
            _lruList.Remove(node);
            _lruList.AddFirst(node);
            return node.Value.Image;
        }
        
        // 加载新图片
        var image = LoadImage(path);
        var newNode = new LinkedListNode<CacheItem>(new CacheItem { Path = path, Image = image });
        
        if (_cache.Count >= _capacity)
        {
            // 移除最久未使用的
            var last = _lruList.Last;
            _lruList.RemoveLast();
            _cache.Remove(last.Value.Path);
        }
        
        _lruList.AddFirst(newNode);
        _cache[path] = newNode;
        
        return image;
    }
    
    private class CacheItem
    {
        public string Path { get; set; }
        public BitmapImage Image { get; set; }
    }
}
```

```csharp
public class MemoryMonitor
{
    private DispatcherTimer _timer;
    
    public void Start()
    {
        _timer = new DispatcherTimer();
        _timer.Interval = TimeSpan.FromSeconds(10);
        _timer.Tick += (s, e) =>
        {
            long memory = GC.GetTotalMemory(false) / 1024 / 1024;

            // 如果内存超过阈值，强制GC
            if (memory > 500)

            }
        };
        _timer.Start();
```

```csharp
public class ParallelDataProcessor

    public async Task<List<ProcessedItem>> ProcessItemsAsync(List<Item> items)
    {
        // 并行处理
        var results = await Task.WhenAll(
            items.Select(item => Task.Run(() => ProcessItem(item)))

        });
    }
```

```csharp
public class RealtimeChart : FrameworkElement
{
    private Queue<Point> _buffer = new Queue<Point>(10000);
    private DispatcherTimer _renderTimer;
    
    public RealtimeChart()
    {
        // 限制渲染频率为30fps（足够流畅）
        _renderTimer = new DispatcherTimer(DispatcherPriority.Render);
        _renderTimer.Interval = TimeSpan.FromMilliseconds(33);
        _renderTimer.Tick += (s, e) => InvalidateVisual();
        _renderTimer.Start();
    }
    
    public void AddData(Point point)
    {
        _buffer.Enqueue(point);
        if (_buffer.Count > 10000)
            _buffer.Dequeue();
        
        // 不立即刷新，等待定时器
    }
    
    protected override void OnRender(DrawingContext dc)
    {
        var points = _buffer.ToArray();
        
        // 使用StreamGeometry（性能最优）
        var geometry = new StreamGeometry();
        using (var ctx = geometry.Open())
        {
            if (points.Length > 0)
            {
                ctx.BeginFigure(points[0], false, false);
                for (int i = 1; i < points.Length; i++)
                {
                    ctx.LineTo(points[i], true, false);
                }
            }
        }
        geometry.Freeze();
        
        dc.DrawGeometry(null, new Pen(Brushes.Blue, 1), geometry);
    }
}
```
