## 1️⃣ C++现代特性

### C++11核心特性

#### 1. 智能指针
```cpp
// shared_ptr - 共享所有权
std::shared_ptr<QWidget> widget = std::make_shared<QWidget>();
std::shared_ptr<QWidget> widget2 = widget;  // 引用计数+1

// unique_ptr - 独占所有权
std::unique_ptr<QFile> file = std::make_unique<QFile>("data.txt");
// unique_ptr<QFile> file2 = file;  // ❌ 编译错误

// unique_ptr 转移所有权
std::unique_ptr<QFile> file2 = std::move(file);  // ✅

// weak_ptr - 弱引用（避免循环引用）
std::weak_ptr<QWidget> weakWidget = widget;
if (auto locked = weakWidget.lock()) {
    // 临时提升为shared_ptr使用
    locked->show();
}
```

**Qt中的应用**:
```cpp
class MyClass : public QObject
{
public:
    // Qt对象树自动管理内存，不一定需要智能指针
    // 但在非QObject类中推荐使用
    std::unique_ptr<DataProcessor> processor;
    
    MyClass(QObject *parent = nullptr)
        : QObject(parent)
        , processor(std::make_unique<DataProcessor>())
    {
    }
};
```

#### 3. 右值引用与移动语义
```cpp
// 左值 vs 右值
int a = 5;       // a是左值
int b = a + 3;   // a+3是右值

// 右值引用
int &&rref = 10;  // 绑定到右值

// 移动构造函数
class MyClass
{
    QString data;
public:
    // 拷贝构造
    MyClass(const MyClass &other) : data(other.data) {
        qDebug() << "Copy constructor";
    }
    
    // 移动构造（C++11）
    MyClass(MyClass &&other) noexcept : data(std::move(other.data)) {
        qDebug() << "Move constructor";
    }

// std::move 强制移动
```

#### 5. nullptr
```cpp
// C++11之前
```

### C++14特性

```cpp
auto lambda = [](auto x, auto y) {
    return x + y;
```

## 5️⃣ 信号槽机制

### 信号槽原理

**核心概念**:
```
信号(Signal)     →     槽(Slot)
发送者(Sender)   →     接收者(Receiver)

特点：
✓ 类型安全
✓ 松耦合
✓ 支持一对多
✓ 支持多对一
✓ 可跨线程
```

### 连接方式

#### 1. 传统字符串连接（Qt4风格）
```cpp
// 不推荐 - 无编译期检查
connect(button, SIGNAL(clicked()), this, SLOT(onButtonClicked()));
```

#### 2. 函数指针连接（Qt5推荐）
```cpp
// 推荐 - 编译期类型检查
QPushButton *button = new QPushButton("Click");

// 连接到成员函数
connect(button, &QPushButton::clicked, this, &MyClass::onButtonClicked);

// 连接到Lambda
connect(button, &QPushButton::clicked, [this]() {
    qDebug() << "Button clicked!";
    updateUI();
});

// 连接带参数的信号
QLineEdit *edit = new QLineEdit();
connect(edit, &QLineEdit::textChanged, [](const QString &text) {
    qDebug() << "Text changed to:" << text;
});
```

#### 3. 连接类型
```cpp
// 自动连接（默认）- 同线程直接调用，跨线程排队
connect(sender, &Sender::signal, receiver, &Receiver::slot);

// 直接连接 - 立即调用，即使跨线程（危险）
connect(sender, &Sender::signal, receiver, &Receiver::slot, Qt::DirectConnection);

// 排队连接 - 事件循环处理（线程安全）
connect(sender, &Sender::signal, receiver, &Receiver::slot, Qt::QueuedConnection);

// 阻塞排队连接 - 等待槽函数执行完毕
connect(sender, &Sender::signal, receiver, &Receiver::slot, Qt::BlockingQueuedConnection);

// 唯一连接 - 防止重复连接
connect(sender, &Sender::signal, receiver, &Receiver::slot, Qt::UniqueConnection);
```

### 自定义信号

```cpp
class MyClass : public QObject
{
    Q_OBJECT
    
signals:
    // 信号声明（只声明，不定义）
    void dataChanged(int value);
    void errorOccurred(const QString &message);
    void progressUpdated(int current, int total);
    
public:
    void doSomething() {
        // 发射信号
        emit dataChanged(42);
        emit progressUpdated(50, 100);
    }
};

// 连接自定义信号
MyClass *obj = new MyClass();
connect(obj, &MyClass::dataChanged, [](int value) {
    qDebug() << "Data changed:" << value;
});

connect(obj, &MyClass::progressUpdated, [](int current, int total) {
    qDebug() << "Progress:" << (current * 100 / total) << "%";
```

### 面试题：信号槽机制

**问题1**: 信号槽机制的实现原理是什么？

**答案**:
```
1. 元对象系统（Meta-Object System）
   ✓ MOC（Meta-Object Compiler）预处理Q_OBJECT宏
   ✓ 生成元对象代码（moc_*.cpp）
   ✓ QMetaObject存储类的元信息

2. 连接存储
   ✓ QObject内部维护连接列表
   ✓ sender记录所有发出的信号连接
   ✓ receiver记录所有接收的槽连接

3. 信号发射
   ✓ emit实际是空宏
   ✓ 信号函数调用QMetaObject::activate()
   ✓ 遍历连接列表，调用对应的槽函数

4. 线程安全
   ✓ 同线程：直接调用（DirectConnection）
   ✓ 跨线程：通过事件队列（QueuedConnection）
   ✓ 自动模式根据线程关系选择
```

**问题2**: 信号槽vs回调函数有什么优势？

**答案**:
```
信号槽优势：
✓ 类型安全（编译期检查）
✓ 松耦合（发送者不需要知道接收者）
✓ 支持一对多、多对一
✓ 线程安全（自动处理跨线程调用）
✓ 自动断开（对象销毁时）

回调函数缺点：
✗ 需要强耦合（持有函数指针/接口）
✗ 单一回调
✗ 需要手动管理生命周期
✗ 跨线程复杂
```

```cpp
class EventFilter : public QObject
{
protected:
    bool eventFilter(QObject *watched, QEvent *event) override {
        if (watched == targetWidget) {
            if (event->type() == QEvent::MouseButtonPress) {
                QMouseEvent *mouseEvent = static_cast<QMouseEvent*>(event);
                qDebug() << "Filtered mouse press at" << mouseEvent->pos();
                return true;  // 事件被过滤，不再传递
            }
        }
        return QObject::eventFilter(watched, event);  // 继续传递
    }
};
```

```cpp
// 自定义Model - 以表格为例
class MyTableModel : public QAbstractTableModel
{
    Q_OBJECT
    
public:
    MyTableModel(QObject *parent = nullptr)
        : QAbstractTableModel(parent)
    {
        // 初始化数据
        m_data = {
            {"John", "25", "New York"},
            {"Alice", "30", "London"},
            {"Bob", "28", "Paris"}
        };
    }
    
    // 必须实现的函数
    int rowCount(const QModelIndex &parent = QModelIndex()) const override {
        Q_UNUSED(parent);
        return m_data.size();
    }
    
    int columnCount(const QModelIndex &parent = QModelIndex()) const override {
        Q_UNUSED(parent);
        return m_data.isEmpty() ? 0 : m_data[0].size();
    }
    
    QVariant data(const QModelIndex &index, int role = Qt::DisplayRole) const override {
        if (!index.isValid())
            return QVariant();
        
        if (role == Qt::DisplayRole || role == Qt::EditRole) {
            return m_data[index.row()][index.column()];
        }
        
        if (role == Qt::BackgroundRole) {
            // 交替行颜色
            return index.row() % 2 == 0 ? QColor(Qt::white) : QColor(240, 240, 240);
        }
        
        if (role == Qt::TextAlignmentRole) {
            return Qt::AlignCenter;
        }
        
        return QVariant();
    }
    
    // 可选：表头
    QVariant headerData(int section, Qt::Orientation orientation, int role) const override {
        if (role == Qt::DisplayRole && orientation == Qt::Horizontal) {
            QStringList headers = {"Name", "Age", "City"};
            return headers[section];
        }
        return QVariant();
    }
    
    // 可选：可编辑
    Qt::ItemFlags flags(const QModelIndex &index) const override {
        if (!index.isValid())
            return Qt::NoItemFlags;
        return Qt::ItemIsEnabled | Qt::ItemIsSelectable | Qt::ItemIsEditable;
    }
    
    bool setData(const QModelIndex &index, const QVariant &value, int role) override {
        if (index.isValid() && role == Qt::EditRole) {
            m_data[index.row()][index.column()] = value.toString();
            emit dataChanged(index, index, {role});
            return true;
        }
        return false;
    }
    
    // 可选：添加/删除行
    bool insertRows(int row, int count, const QModelIndex &parent = QModelIndex()) override {
        beginInsertRows(parent, row, row + count - 1);
        for (int i = 0; i < count; ++i) {
            m_data.insert(row, {"", "", ""});
        }
        endInsertRows();
        return true;
    }
    
    bool removeRows(int row, int count, const QModelIndex &parent = QModelIndex()) override {
        beginRemoveRows(parent, row, row + count - 1);
        for (int i = 0; i < count; ++i) {
            m_data.removeAt(row);
        }
        endRemoveRows();
        return true;
    }
    
private:
    QVector<QVector<QString>> m_data;
};
```

```cpp
{
    QPixmap m_cache;
    bool m_cacheValid = false;
    
protected:
    void paintEvent(QPaintEvent *event) override {
        if (!m_cacheValid) {
            m_cache = QPixmap(size());
            QPainter cachePainter(&m_cache);
            // 绘制到缓存...
            m_cacheValid = true;
        }
        
        QPainter painter(this);
        painter.drawPixmap(0, 0, m_cache);
    }

    }
```
