## 12.1 Qt 如何实现跨平台？

### 答案

Qt 通过**平台抽象层（Platform Abstraction Layer, PAL）**实现跨平台。

### 核心机制

```
┌─────────────────────────────────┐
│      Qt 应用程序代码            │
├─────────────────────────────────┤
│      Qt API 层（统一接口）       │
├─────────────────────────────────┤
│    QPA (Qt Platform Abstraction) │
├──────┬──────────┬────────┬──────┤
│Windows│  macOS  │ Linux  │ ...  │
└──────┴──────────┴────────┴──────┘
```

### QPA（Qt Platform Abstraction）

Qt 为每个平台提供了**插件化的平台实现**：

| 平台 | QPA 插件 | 说明 |
|------|---------|------|
| Windows | qwindows | Win32 API 实现 |
| macOS | qcocoa | Cocoa API 实现 |
| Linux | qxcb | X11 实现 |
| Linux | qwayland | Wayland 实现 |
| Android | qandroid | Android 实现 |
| iOS | qios | iOS 实现 |

---

### 跨平台代码示例

```cpp
// 同一套代码，在所有平台运行
#include <QApplication>
#include <QMainWindow>
#include <QPushButton>

int main(int argc, char* argv[]) {
    QApplication app(argc, argv);
    
    QMainWindow window;
    QPushButton* button = new QPushButton("点击我", &window);
    window.setCentralWidget(button);
    window.show();
    
    return app.exec();
}

// 在 Windows、macOS、Linux 上外观不同，但代码相同
```

## 12.2 如何处理平台特定代码？

### 答案

使用**条件编译**和 Qt 的平台检测宏。

---

### 1. 平台检测宏

| 宏 | 平台 |
|---|------|
| Q_OS_WIN | Windows |
| Q_OS_MACOS | macOS |
| Q_OS_LINUX | Linux |
| Q_OS_ANDROID | Android |
| Q_OS_IOS | iOS |
| Q_OS_UNIX | Unix 系列（Linux、macOS、BSD） |

---

### 2. 条件编译示例

```cpp
#include <QDebug>

void showPlatformInfo() {
#ifdef Q_OS_WIN
    qDebug() << "当前平台: Windows";
    // Windows 特定代码
    #include <windows.h>
    MessageBoxA(nullptr, "Hello", "Windows", MB_OK);
#elif defined(Q_OS_MACOS)
    qDebug() << "当前平台: macOS";
    // macOS 特定代码
#elif defined(Q_OS_LINUX)
    qDebug() << "当前平台: Linux";
    // Linux 特定代码
#else
    qDebug() << "当前平台: 未知";
#endif
}
```

---

### 3. 平台特定文件路径

```cpp
QString getConfigPath() {
#ifdef Q_OS_WIN
    // Windows: C:/Users/Username/AppData/Local/MyApp
    return QStandardPaths::writableLocation(QStandardPaths::AppLocalDataLocation);
#elif defined(Q_OS_MACOS)
    // macOS: /Users/Username/Library/Application Support/MyApp
    return QStandardPaths::writableLocation(QStandardPaths::AppDataLocation);
#else
    // Linux: /home/username/.local/share/MyApp
    return QStandardPaths::writableLocation(QStandardPaths::AppDataLocation);
#endif
}

// 推荐：使用 Qt 提供的跨平台 API
QString getConfigPath() {
    return QStandardPaths::writableLocation(QStandardPaths::AppDataLocation);
}
```

---

### 4. .pro 文件条件编译

```qmake
# 公共配置
QT += core gui widgets
TARGET = MyApp

# Windows 特定
win32 {
    RC_FILE = app.rc
    LIBS += -luser32 -lgdi32
    DEFINES += WIN32_APP
}

# macOS 特定
macx {
    ICON = app.icns
    QMAKE_INFO_PLIST = Info.plist
    DEFINES += MACOS_APP
}

# Linux 特定
unix:!macx {
    LIBS += -lX11
    DEFINES += LINUX_APP
}
```

---

### 5. 平台特定源文件

```qmake
# 公共源文件
SOURCES += main.cpp \
           mainwindow.cpp

# 平台特定源文件
win32: SOURCES += platform_win.cpp
macx: SOURCES += platform_mac.cpp
unix:!macx: SOURCES += platform_linux.cpp
```

```cpp
// platform_win.cpp (仅 Windows 编译)
#include "platform.h"
#include <windows.h>

void Platform::showNotification(const QString& message) {
    MessageBoxW(nullptr, message.toStdWString().c_str(), L"通知", MB_OK);
}

// platform_mac.cpp (仅 macOS 编译)
#include "platform.h"
#import <Cocoa/Cocoa.h>

void Platform::showNotification(const QString& message) {
    NSString* msg = message.toNSString();
    NSAlert* alert = [[NSAlert alloc] init];
    [alert setMessageText:msg];
    [alert runModal];
}
```

## 12.3 如何处理 Hi-DPI 显示？

### 答案

Qt 5.6+ 提供了**自动 Hi-DPI 支持**，需要正确配置。

---

### 1. 启用 Hi-DPI 支持

```cpp
#include <QApplication>

int main(int argc, char* argv[]) {
    // Qt 5.6 - Qt 5.13
    QApplication::setAttribute(Qt::AA_EnableHighDpiScaling);
    QApplication::setAttribute(Qt::AA_UseHighDpiPixmaps);
    
    // Qt 5.14+（自动启用，可选配置）
    // QGuiApplication::setHighDpiScaleFactorRoundingPolicy(
    //     Qt::HighDpiScaleFactorRoundingPolicy::PassThrough);
    
    QApplication app(argc, argv);
    
    // ...
    
    return app.exec();
}
```

---

### 2. 提供多分辨率图片

```
resources/
├── icon.png          (100x100, 1x)
├── icon@2x.png       (200x200, 2x, Retina)
├── icon@3x.png       (300x300, 3x)
└── icon.svg          (矢量图，推荐)
```

```cpp
// Qt 自动根据 DPI 选择合适的图片
QIcon icon(":/resources/icon.png");
button->setIcon(icon);

// 推荐：使用 SVG（自动缩放）
QIcon svgIcon(":/resources/icon.svg");
```

---

### 3. 使用逻辑像素而非物理像素

```cpp
// ❌ 错误：硬编码物理像素
widget->setFixedSize(200, 100);  // 在 Hi-DPI 屏幕上太小

// ✅ 正确：使用逻辑像素（Qt 自动缩放）
widget->setFixedSize(200, 100);  // Qt 会自动处理

// 或使用字体大小相对布局
QFont font;
font.setPointSize(12);  // 点大小，自动缩放
label->setFont(font);
```

---

### 4. 检测屏幕 DPI

```cpp
#include <QScreen>
#include <QGuiApplication>

void printScreenInfo() {
    QScreen* screen = QGuiApplication::primaryScreen();
    
    qreal dpr = screen->devicePixelRatio();
    qDebug() << "设备像素比:" << dpr;  // 1.0 (普通), 2.0 (Retina)
    
    QSize logicalSize = screen->size();
    QSize physicalSize = screen->physicalSize();
    
    qDebug() << "逻辑分辨率:" << logicalSize;
    qDebug() << "物理分辨率:" << physicalSize;
}
```

---

### 5. 自绘控件的 Hi-DPI 处理

```cpp
class HighDpiWidget : public QWidget {
protected:
    void paintEvent(QPaintEvent*) override {
        QPainter painter(this);
        
        // 获取设备像素比
        qreal dpr = devicePixelRatio();
        
        // 创建高分辨率 QPixmap
        QPixmap pixmap(width() * dpr, height() * dpr);
        pixmap.setDevicePixelRatio(dpr);
        pixmap.fill(Qt::transparent);
        
        QPainter pixmapPainter(&pixmap);
        // 绘制内容到 pixmap
        pixmapPainter.drawEllipse(10, 10, 80, 80);
        
        // 绘制到窗口
        painter.drawPixmap(0, 0, pixmap);
    }
};
```

## 12.4 Windows、macOS、Linux 平台有哪些差异？

### 答案

三大平台在**文件系统、窗口行为、键盘快捷键**等方面存在差异。

---

### 1. 文件系统差异

| 特性 | Windows | macOS | Linux |
|------|---------|-------|-------|
| 路径分隔符 | `\` | `/` | `/` |
| 根目录 | 多个（C:\, D:\） | `/` | `/` |
| 文件名大小写 | 不敏感 | 不敏感（可配置） | 敏感 |
| 换行符 | `\r\n` (CRLF) | `\n` (LF) | `\n` (LF) |

```cpp
// ❌ 错误：硬编码路径分隔符
QString path = "C:\\Users\\Me\\file.txt";  // 只在 Windows 工作

// ✅ 正确：使用 QDir
QString path = QDir::homePath() + QDir::separator() + "file.txt";

// 更好：使用 Qt 路径 API
QString path = QDir::home().filePath("file.txt");
// Windows: C:/Users/Me/file.txt
// macOS:   /Users/Me/file.txt
// Linux:   /home/me/file.txt
```

---

### 2. 窗口行为差异

```cpp
// macOS: 关闭窗口不退出应用
#ifdef Q_OS_MACOS
QApplication::setQuitOnLastWindowClosed(false);

// 菜单栏在屏幕顶部，不在窗口内
QMenuBar* menuBar = new QMenuBar(nullptr);  // 全局菜单栏
#endif

// Windows/Linux: 关闭窗口退出应用
#ifndef Q_OS_MACOS
QApplication::setQuitOnLastWindowClosed(true);
#endif
```

---

### 3. 键盘快捷键差异

| 功能 | Windows/Linux | macOS |
|------|--------------|-------|
| 复制 | Ctrl+C | Cmd+C |
| 粘贴 | Ctrl+V | Cmd+V |
| 撤销 | Ctrl+Z | Cmd+Z |
| 全选 | Ctrl+A | Cmd+A |

```cpp
// ❌ 错误：硬编码快捷键
QAction* copyAction = new QAction("复制");
copyAction->setShortcut(QKeySequence("Ctrl+C"));  // macOS 上不对

// ✅ 正确：使用标准快捷键
copyAction->setShortcut(QKeySequence::Copy);  // 自动适配平台
// Windows: Ctrl+C
// macOS:   Cmd+C

// 其他标准快捷键
QKeySequence::New          // 新建
QKeySequence::Open         // 打开
QKeySequence::Save         // 保存
QKeySequence::Quit         // 退出
QKeySequence::Undo         // 撤销
QKeySequence::Redo         // 重做
```

---

### 4. 原生外观

```cpp
// Qt 默认使用平台原生外观
// Windows: Windows 样式
// macOS:   macOS 样式
// Linux:   Fusion 样式

// 可以手动设置样式
QApplication::setStyle("Fusion");  // 跨平台统一样式

// 获取可用样式
QStringList styles = QStyleFactory::keys();
qDebug() << styles;  // Windows, WindowsVista, Fusion, ...
```

---

### 5. 应用图标

```qmake
# Windows (.ico, 256x256)
win32: RC_ICONS = app.ico

# macOS (.icns, 1024x1024)
macx: ICON = app.icns

# Linux (.png, 512x512)
unix:!macx {
    isEmpty(PREFIX) { PREFIX = /usr/local }
    target.path = $$PREFIX/bin
    icon.path = $$PREFIX/share/icons/hicolor/512x512/apps
    icon.files = app.png
    INSTALLS += target icon
}
```

---

### 6. 系统托盘图标

```cpp
#include <QSystemTrayIcon>

QSystemTrayIcon* tray = new QSystemTrayIcon(this);

#ifdef Q_OS_MACOS
// macOS: 使用黑白图标（自动适配深色模式）
tray->setIcon(QIcon(":/tray_template.png"));
#else
// Windows/Linux: 使用彩色图标
tray->setIcon(QIcon(":/tray_color.png"));
#endif

tray->show();
```

---

### 7. 文件对话框

```cpp
// 使用原生文件对话框
QString file = QFileDialog::getOpenFileName(
    this, "打开文件", "", "文本文件 (*.txt)");

// 强制使用 Qt 文件对话框（跨平台一致）
QString file = QFileDialog::getOpenFileName(
    this, "打开文件", "", "文本文件 (*.txt)",
    nullptr, QFileDialog::DontUseNativeDialog);
```

## 本章小结

### 核心知识点

1. **跨平台机制**
   - QPA（Qt Platform Abstraction）
   - 平台抽象层隔离平台差异

2. **平台特定代码**
   - 条件编译宏（Q_OS_WIN、Q_OS_MACOS、Q_OS_LINUX）
   - .pro 文件平台配置
   - 平台特定源文件

3. **Hi-DPI 支持**
   - 启用 Qt::AA_EnableHighDpiScaling
   - 提供多分辨率图片（@2x、@3x、.svg）
   - 使用逻辑像素

4. **平台差异**
   - 文件系统（路径分隔符、大小写）
   - 窗口行为（macOS 特殊）
   - 键盘快捷键（Ctrl vs Cmd）
   - 原生外观和对话框

### 最佳实践

- ✅ 优先使用 Qt 跨平台 API
- ✅ 避免硬编码路径分隔符
- ✅ 使用标准快捷键（QKeySequence）
- ✅ 提供矢量图标（.svg）
- ✅ 测试所有目标平台
- ✅ 使用 QStandardPaths 获取系统路径
