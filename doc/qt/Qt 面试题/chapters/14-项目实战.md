## 14.1 如何集成第三方库？

### 答案

Qt 项目集成第三方库有**多种方式**，根据库的类型选择合适的方法。

---

### 1. 集成头文件库（Header-Only）

```qmake
# .pro 文件
INCLUDEPATH += $$PWD/third_party/json/include
```

```cpp
// 使用
#include <nlohmann/json.hpp>

using json = nlohmann::json;

json j;
j["name"] = "张三";
j["age"] = 25;

std::string str = j.dump();
```

---

### 2. 集成静态库（.a / .lib）

```qmake
# .pro 文件

# 添加头文件路径
INCLUDEPATH += $$PWD/third_party/openssl/include

# 链接静态库
win32 {
    LIBS += -L$$PWD/third_party/openssl/lib -lssl -lcrypto
}

unix {
    LIBS += -L$$PWD/third_party/openssl/lib -lssl -lcrypto
}

macx {
    LIBS += -L$$PWD/third_party/openssl/lib -lssl -lcrypto
}
```

---

### 3. 集成动态库（.so / .dll / .dylib）

```qmake
# .pro 文件
INCLUDEPATH += $$PWD/third_party/ffmpeg/include

win32 {
    LIBS += -L$$PWD/third_party/ffmpeg/lib -lavcodec -lavformat -lavutil
    
    # 复制 DLL 到输出目录
    QMAKE_POST_LINK += $$quote(xcopy /Y /I \"$$PWD/third_party/ffmpeg/bin/*.dll\" \"$$OUT_PWD/release\")
}

unix:!macx {
    LIBS += -L$$PWD/third_party/ffmpeg/lib -lavcodec -lavformat -lavutil
    
    # 设置运行时库搜索路径
    QMAKE_RPATHDIR += $$PWD/third_party/ffmpeg/lib
}

macx {
    LIBS += -L$$PWD/third_party/ffmpeg/lib -lavcodec -lavformat -lavutil
    QMAKE_RPATHDIR += $$PWD/third_party/ffmpeg/lib
}
```

---

### 4. 使用 pkg-config

```qmake
# .pro 文件
QT += core gui widgets

# 使用 pkg-config 查找库
CONFIG += link_pkgconfig
PKGCONFIG += opencv4

# 或手动指定
unix {
    PKGCONFIG += opencv4
}
```

---

### 5. CMake 集成第三方库

```cmake
# CMakeLists.txt
cmake_minimum_required(VERSION 3.16)
project(MyApp)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

find_package(Qt6 COMPONENTS Core Gui Widgets REQUIRED)

# 查找第三方库
find_package(OpenCV REQUIRED)
include_directories(${OpenCV_INCLUDE_DIRS})

# 添加源文件
add_executable(MyApp
    main.cpp
    mainwindow.cpp
)

# 链接 Qt 和第三方库
target_link_libraries(MyApp
    Qt6::Core
    Qt6::Gui
    Qt6::Widgets
    ${OpenCV_LIBS}
)
```

---

### 6. 子项目方式（SUBDIRS）

```qmake
# myapp.pro (主项目)
TEMPLATE = subdirs
SUBDIRS += \
    third_party/mylib \
    src

src.depends = third_party/mylib

# third_party/mylib/mylib.pro
TEMPLATE = lib
CONFIG += staticlib
SOURCES += mylib.cpp
HEADERS += mylib.h

# src/src.pro
TEMPLATE = app
QT += core gui widgets

LIBS += -L$$OUT_PWD/../third_party/mylib -lmylib
INCLUDEPATH += $$PWD/../third_party/mylib

SOURCES += main.cpp
```

## 14.2 如何编写单元测试？

### 答案

Qt 提供了 **Qt Test 模块**用于单元测试。

---

### 1. 创建测试项目

```qmake
# tests.pro
QT += core testlib
QT -= gui

CONFIG += qt console warn_on depend_includepath testcase
CONFIG -= app_bundle

TEMPLATE = app

SOURCES += test_calculator.cpp
```

---

### 2. 编写测试类

```cpp
// test_calculator.cpp
#include <QtTest>

// 被测试的类
class Calculator {
public:
    int add(int a, int b) { return a + b; }
    int subtract(int a, int b) { return a - b; }
    int multiply(int a, int b) { return a * b; }
    double divide(int a, int b) { 
        if (b == 0) throw std::runtime_error("Division by zero");
        return static_cast<double>(a) / b; 
    }
};

// 测试类
class TestCalculator : public QObject {
    Q_OBJECT
    
private slots:
    // 初始化（每个测试前执行）
    void init() {
        m_calc = new Calculator();
    }
    
    // 清理（每个测试后执行）
    void cleanup() {
        delete m_calc;
    }
    
    // 测试用例
    void testAdd() {
        QCOMPARE(m_calc->add(2, 3), 5);
        QCOMPARE(m_calc->add(-1, 1), 0);
        QCOMPARE(m_calc->add(0, 0), 0);
    }
    
    void testSubtract() {
        QCOMPARE(m_calc->subtract(5, 3), 2);
        QCOMPARE(m_calc->subtract(1, 1), 0);
        QCOMPARE(m_calc->subtract(0, 5), -5);
    }
    
    void testMultiply() {
        QCOMPARE(m_calc->multiply(2, 3), 6);
        QCOMPARE(m_calc->multiply(-2, 3), -6);
        QCOMPARE(m_calc->multiply(0, 100), 0);
    }
    
    void testDivide() {
        QCOMPARE(m_calc->divide(6, 2), 3.0);
        QCOMPARE(m_calc->divide(7, 2), 3.5);
        
        // 测试异常
        QVERIFY_EXCEPTION_THROWN(m_calc->divide(5, 0), std::runtime_error);
    }
    
    // 数据驱动测试
    void testAdd_data() {
        QTest::addColumn<int>("a");
        QTest::addColumn<int>("b");
        QTest::addColumn<int>("expected");
        
        QTest::newRow("positive") << 2 << 3 << 5;
        QTest::newRow("negative") << -1 << 1 << 0;
        QTest::newRow("zero") << 0 << 0 << 0;
    }
    
    void testAdd2() {
        QFETCH(int, a);
        QFETCH(int, b);
        QFETCH(int, expected);
        
        QCOMPARE(m_calc->add(a, b), expected);
    }
    
private:
    Calculator* m_calc;
};

// 运行测试
QTEST_MAIN(TestCalculator)
#include "test_calculator.moc"
```

---

### 3. 常用断言宏

| 宏 | 说明 |
|---|------|
| QCOMPARE(actual, expected) | 比较相等 |
| QVERIFY(condition) | 验证条件为真 |
| QVERIFY2(condition, message) | 带消息的验证 |
| QTEST(actual, testdata) | 数据驱动测试 |
| QFAIL(message) | 标记测试失败 |
| QSKIP(message) | 跳过测试 |
| QEXPECT_FAIL(tag, message, mode) | 预期失败 |
| QVERIFY_EXCEPTION_THROWN(expr, exception) | 验证异常 |

---

### 4. 测试 Qt 控件

```cpp
class TestWidget : public QObject {
    Q_OBJECT
    
private slots:
    void testButtonClick() {
        QPushButton button("Click Me");
        
        // 信号间谍
        QSignalSpy spy(&button, &QPushButton::clicked);
        
        // 模拟点击
        QTest::mouseClick(&button, Qt::LeftButton);
        
        // 验证信号发射
        QCOMPARE(spy.count(), 1);
    }
    
    void testLineEdit() {
        QLineEdit lineEdit;
        
        // 模拟键盘输入
        QTest::keyClicks(&lineEdit, "Hello");
        QCOMPARE(lineEdit.text(), "Hello");
        
        // 模拟快捷键
        QTest::keyClick(&lineEdit, Qt::Key_A, Qt::ControlModifier);  // Ctrl+A
    }
};
```

## 14.3 如何调试性能问题？

### 答案

性能问题调试需要**分析瓶颈**，使用多种工具定位问题。

---

### 1. QElapsedTimer 计时

```cpp
#include <QElapsedTimer>
#include <QDebug>

void slowFunction() {
    QElapsedTimer timer;
    timer.start();
    
    // 耗时操作
    for (int i = 0; i < 1000000; ++i) {
        // ...
    }
    
    qDebug() << "耗时:" << timer.elapsed() << "ms";
}
```

---

### 2. Qt Creator Profiler

```
菜单 → 分析 → 性能分析器
    → CPU 使用率
    → 内存分配
    → 函数调用图
```

**分析报告包括**：
- 热点函数（耗时最多）
- 调用次数
- 调用栈
- 内存分配

---

### 3. 性能日志宏

```cpp
// performance.h
#ifndef PERFORMANCE_H
#define PERFORMANCE_H

#include <QElapsedTimer>
#include <QDebug>

class PerformanceTimer {
public:
    explicit PerformanceTimer(const char* name) : m_name(name) {
        m_timer.start();
    }
    
    ~PerformanceTimer() {
        qDebug() << "[PERF]" << m_name << ":" << m_timer.elapsed() << "ms";
    }
    
private:
    const char* m_name;
    QElapsedTimer m_timer;
};

#define PERF_TIMER(name) PerformanceTimer __perf_timer__(name)

#endif
```

```cpp
// 使用
void myFunction() {
    PERF_TIMER("myFunction");
    
    // 函数代码
    // ...
    
    // 离开作用域时自动输出耗时
}
```

### 6. 绘制性能问题调试

```cpp
class ProfiledWidget : public QWidget {
protected:
    void paintEvent(QPaintEvent*) override {
        QElapsedTimer timer;
        timer.start();
        
        QPainter painter(this);
        
        // 绘制代码
        drawComplexShape(&painter);
        
        qint64 elapsed = timer.elapsed();
        if (elapsed > 16) {  // 超过 16ms（60fps）
            qDebug() << "绘制过慢:" << elapsed << "ms";
        }
    }
};
```

## 14.4 如何处理崩溃和日志？

### 答案

崩溃处理需要**捕获异常、记录日志、生成 dump 文件**。

---

### 1. 安装消息处理器

```cpp
#include <QDebug>
#include <QDateTime>
#include <QFile>
#include <QTextStream>

void messageHandler(QtMsgType type, const QMessageLogContext& context,
                   const QString& msg) {
    QString level;
    switch (type) {
        case QtDebugMsg: level = "DEBUG"; break;
        case QtInfoMsg: level = "INFO"; break;
        case QtWarningMsg: level = "WARNING"; break;
        case QtCriticalMsg: level = "CRITICAL"; break;
        case QtFatalMsg: level = "FATAL"; break;
    }
    
    QString timestamp = QDateTime::currentDateTime().toString("yyyy-MM-dd hh:mm:ss");
    QString logMessage = QString("[%1] [%2] %3 (%4:%5, %6)")
        .arg(timestamp)
        .arg(level)
        .arg(msg)
        .arg(context.file)
        .arg(context.line)
        .arg(context.function);
    
    // 输出到控制台
    QTextStream(stderr) << logMessage << "\n";
    
    // 写入日志文件
    QFile file("app.log");
    if (file.open(QIODevice::WriteOnly | QIODevice::Append)) {
        QTextStream stream(&file);
        stream << logMessage << "\n";
        file.close();
    }
    
    // Fatal 错误后终止程序
    if (type == QtFatalMsg) {
        abort();
    }
}

int main(int argc, char* argv[]) {
    qInstallMessageHandler(messageHandler);
    
    QApplication app(argc, argv);
    
    qDebug() << "应用启动";
    qWarning() << "警告消息";
    qCritical() << "严重错误";
    
    return app.exec();
}
```

---

### 2. 异常捕获

```cpp
int main(int argc, char* argv[]) {
    try {
        QApplication app(argc, argv);
        
        MainWindow window;
        window.show();
        
        return app.exec();
    }
    catch (const std::exception& e) {
        qCritical() << "未捕获异常:" << e.what();
        QMessageBox::critical(nullptr, "错误", 
            QString("程序发生错误: %1").arg(e.what()));
        return -1;
    }
    catch (...) {
        qCritical() << "未知异常";
        QMessageBox::critical(nullptr, "错误", "程序发生未知错误");
        return -1;
    }
}
```

---

### 3. 生成 Dump 文件（Windows）

```cpp
#ifdef Q_OS_WIN
#include <windows.h>
#include <dbghelp.h>

LONG WINAPI exceptionHandler(EXCEPTION_POINTERS* exceptionInfo) {
    QString dumpPath = QCoreApplication::applicationDirPath() + "/crash.dmp";
    
    HANDLE hFile = CreateFileW(
        dumpPath.toStdWString().c_str(),
        GENERIC_WRITE,
        0,
        NULL,
        CREATE_ALWAYS,
        FILE_ATTRIBUTE_NORMAL,
        NULL
    );
    
    if (hFile != INVALID_HANDLE_VALUE) {
        MINIDUMP_EXCEPTION_INFORMATION mdei;
        mdei.ThreadId = GetCurrentThreadId();
        mdei.ExceptionPointers = exceptionInfo;
        mdei.ClientPointers = FALSE;
        
        MiniDumpWriteDump(
            GetCurrentProcess(),
            GetCurrentProcessId(),
            hFile,
            MiniDumpNormal,
            &mdei,
            NULL,
            NULL
        );
        
        CloseHandle(hFile);
        
        qCritical() << "崩溃 dump 文件已生成:" << dumpPath;
    }
    
    return EXCEPTION_EXECUTE_HANDLER;
}

int main(int argc, char* argv[]) {
    SetUnhandledExceptionFilter(exceptionHandler);
    
    QApplication app(argc, argv);
    // ...
    return app.exec();
}
#endif
```

### 4. 日志轮转

```cpp
class LogRotator {
public:
    static void setupLogging() {
        QString logPath = QStandardPaths::writableLocation(
            QStandardPaths::AppDataLocation) + "/logs";
        QDir().mkpath(logPath);
        
        // 检查日志文件大小，超过 10MB 则轮转
        QString currentLog = logPath + "/app.log";
        QFileInfo info(currentLog);
        
        if (info.size() > 10 * 1024 * 1024) {  // 10MB
            // 重命名旧日志
            QString oldLog = logPath + "/app_" + 
                QDateTime::currentDateTime().toString("yyyyMMdd_hhmmss") + ".log";
            QFile::rename(currentLog, oldLog);
            
            // 删除超过 7 天的日志
            QDir logDir(logPath);
            QFileInfoList files = logDir.entryInfoList(
                QStringList() << "app_*.log", QDir::Files, QDir::Time);
            
            for (int i = 7; i < files.size(); ++i) {
                QFile::remove(files[i].absoluteFilePath());
            }
        }
    }
};
```

## 14.5 如何构建和打包应用？

### 答案

不同平台的打包方式不同。

---

### 1. Windows 打包

```bash
# 1. 编译 Release 版本
qmake CONFIG+=release
nmake  # 或 mingw32-make

# 2. 使用 windeployqt 收集依赖
windeployqt.exe --release myapp.exe

# 3. 创建安装程序（使用 Inno Setup 或 NSIS）
```

**Inno Setup 脚本示例**：

```iss
[Setup]
AppName=MyApp
AppVersion=1.0
DefaultDirName={pf}\MyApp
OutputBaseFilename=MyAppSetup

[Files]
Source: "release\*"; DestDir: "{app}"; Flags: recursesubdirs

[Icons]
Name: "{commondesktop}\MyApp"; Filename: "{app}\myapp.exe"
```

---

### 2. macOS 打包

```bash
# 1. 编译 Release 版本
qmake CONFIG+=release
make

# 2. 使用 macdeployqt 创建 .app
macdeployqt myapp.app

# 3. 创建 DMG 安装包
hdiutil create -volname "MyApp" -srcfolder myapp.app -ov -format UDZO MyApp.dmg

# 4. 代码签名（可选）
codesign --deep --force --verify --verbose --sign "Developer ID" myapp.app
```

---

### 3. Linux 打包

```bash
# 1. 编译 Release 版本
qmake CONFIG+=release
make

# 2. 收集依赖（使用 linuxdeployqt）
linuxdeployqt myapp -appimage

# 3. 或创建 DEB 包
mkdir -p myapp_1.0/DEBIAN
mkdir -p myapp_1.0/usr/bin
mkdir -p myapp_1.0/usr/share/applications

# 创建 control 文件
cat > myapp_1.0/DEBIAN/control <<EOF
Package: myapp
Version: 1.0
Architecture: amd64
Maintainer: Your Name <email@example.com>
Description: My Qt Application
EOF

# 复制文件
cp myapp myapp_1.0/usr/bin/

# 构建 DEB 包
dpkg-deb --build myapp_1.0
```
