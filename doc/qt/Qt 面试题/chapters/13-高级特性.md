## 13.1 如何实现 Qt 插件系统？

### 答案

Qt 提供了**QPluginLoader** 和**接口宏**机制实现插件系统。

---

### 插件系统架构

```
┌──────────────┐
│  主应用程序   │
├──────────────┤
│QPluginLoader │
└──────┬───────┘
       │ 加载
   ┌───┴───┬───────┬───────┐
   │插件A  │插件B  │插件C  │
   └───────┴───────┴───────┘
```

---

### 1. 定义插件接口

```cpp
// plugin_interface.h
#ifndef PLUGIN_INTERFACE_H
#define PLUGIN_INTERFACE_H

#include <QString>
#include <QtPlugin>

class PluginInterface {
public:
    virtual ~PluginInterface() = default;
    
    virtual QString name() const = 0;
    virtual QString version() const = 0;
    virtual void initialize() = 0;
    virtual void execute() = 0;
};

// 声明接口
Q_DECLARE_INTERFACE(PluginInterface, "com.example.PluginInterface/1.0")

#endif
```

---

### 2. 实现插件

```cpp
// my_plugin.h
#ifndef MY_PLUGIN_H
#define MY_PLUGIN_H

#include "plugin_interface.h"
#include <QObject>

class MyPlugin : public QObject, public PluginInterface {
    Q_OBJECT
    Q_PLUGIN_METADATA(IID "com.example.PluginInterface/1.0")
    Q_INTERFACES(PluginInterface)
    
public:
    QString name() const override {
        return "MyPlugin";
    }
    
    QString version() const override {
        return "1.0.0";
    }
    
    void initialize() override {
        qDebug() << "MyPlugin initialized";
    }
    
    void execute() override {
        qDebug() << "MyPlugin executed";
    }
};

#endif

// my_plugin.cpp
#include "my_plugin.h"
```

---

### 3. 插件 .pro 文件

```qmake
# my_plugin.pro
TEMPLATE = lib
CONFIG += plugin
QT += core gui widgets

TARGET = my_plugin
DESTDIR = ../plugins

HEADERS += my_plugin.h \
           ../plugin_interface.h

SOURCES += my_plugin.cpp
```

---

### 4. 加载插件

```cpp
// main.cpp
#include <QCoreApplication>
#include <QPluginLoader>
#include <QDir>
#include <QDebug>
#include "plugin_interface.h"

void loadPlugins() {
    QDir pluginsDir(qApp->applicationDirPath());
    pluginsDir.cd("plugins");
    
    // 遍历插件目录
    foreach (QString fileName, pluginsDir.entryList(QDir::Files)) {
        QString filePath = pluginsDir.absoluteFilePath(fileName);
        
        QPluginLoader loader(filePath);
        QObject* plugin = loader.instance();
        
        if (plugin) {
            PluginInterface* interface = qobject_cast<PluginInterface*>(plugin);
            if (interface) {
                qDebug() << "加载插件:" << interface->name()
                         << "版本:" << interface->version();
                interface->initialize();
                interface->execute();
            }
        } else {
            qDebug() << "加载失败:" << loader.errorString();
        }
    }
}

int main(int argc, char* argv[]) {
    QCoreApplication app(argc, argv);
    
    loadPlugins();
    
    return app.exec();
}
```

---

### 5. 插件管理器示例

```cpp
class PluginManager : public QObject {
    Q_OBJECT
public:
    static PluginManager& instance() {
        static PluginManager instance;
        return instance;
    }
    
    void loadPlugins(const QString& path) {
        QDir dir(path);
        foreach (QString fileName, dir.entryList(QDir::Files)) {
            QString filePath = dir.absoluteFilePath(fileName);
            
            QPluginLoader* loader = new QPluginLoader(filePath, this);
            QObject* plugin = loader->instance();
            
            if (plugin) {
                PluginInterface* interface = qobject_cast<PluginInterface*>(plugin);
                if (interface) {
                    m_plugins.append(interface);
                    emit pluginLoaded(interface->name());
                }
            }
        }
    }
    
    QList<PluginInterface*> plugins() const {
        return m_plugins;
    }
    
    PluginInterface* plugin(const QString& name) const {
        for (PluginInterface* p : m_plugins) {
            if (p->name() == name) {
                return p;
            }
        }
        return nullptr;
    }
    
signals:
    void pluginLoaded(const QString& name);
    
private:
    PluginManager() = default;
    QList<PluginInterface*> m_plugins;
};
```

## 13.2 如何使用 Qt 资源系统？

### 答案

Qt 资源系统通过 **.qrc 文件**将资源编译到可执行文件中。

---

### 1. 创建资源文件

```xml
<!-- resources.qrc -->
<!DOCTYPE RCC>
<RCC version="1.0">
    <qresource>
        <file>images/logo.png</file>
        <file>images/icon.png</file>
        <file>styles/style.qss</file>
        <file>fonts/custom.ttf</file>
    </qresource>
    
    <!-- 带前缀的资源 -->
    <qresource prefix="/icons">
        <file alias="save.png">images/save_icon.png</file>
        <file alias="open.png">images/open_icon.png</file>
    </qresource>
</RCC>
```

---

### 2. 在 .pro 中添加资源

```qmake
RESOURCES += resources.qrc
```

---

### 3. 使用资源

```cpp
// 加载图片
QPixmap pixmap(":/images/logo.png");
label->setPixmap(pixmap);

// 加载图标
QIcon icon(":/icons/save.png");  // 使用别名
button->setIcon(icon);

// 加载样式表
QFile file(":/styles/style.qss");
if (file.open(QIODevice::ReadOnly)) {
    QString styleSheet = file.readAll();
    qApp->setStyleSheet(styleSheet);
    file.close();
}

// 加载字体
int fontId = QFontDatabase::addApplicationFont(":/fonts/custom.ttf");
QStringList fontFamilies = QFontDatabase::applicationFontFamilies(fontId);
if (!fontFamilies.isEmpty()) {
    QFont font(fontFamilies.at(0));
    qApp->setFont(font);
}
```

---

### 4. 资源别名和前缀

```xml
<qresource prefix="/images">
    <file alias="logo">company_logo_2024.png</file>
</qresource>
```

```cpp
// 访问：:/images/logo
QPixmap pixmap(":/images/logo");
```

---

### 5. 动态加载资源文件

```cpp
// 编译外部资源文件
// rcc -binary resources.qrc -o resources.rcc

// 运行时加载
bool success = QResource::registerResource("resources.rcc");
if (success) {
    QPixmap pixmap(":/images/logo.png");
}

// 卸载资源
QResource::unregisterResource("resources.rcc");
```

---

### 6. 资源压缩

```xml
<!-- 自动压缩资源 -->
<qresource>
    <file compression="9">large_image.png</file>  <!-- 压缩级别 0-9 -->
    <file compression="0">already_compressed.jpg</file>  <!-- 不压缩 -->
</qresource>
```

## 13.3 如何实现样式表（QSS）自定义？

### 答案

Qt 样式表（QSS）类似 CSS，可以自定义控件外观。

---

### 1. 基本语法

```cpp
// 设置单个控件样式
button->setStyleSheet("background-color: blue; color: white;");

// 设置整个应用样式
qApp->setStyleSheet(R"(
    QPushButton {
        background-color: #4CAF50;
        color: white;
        border: none;
        padding: 10px 20px;
        border-radius: 5px;
    }
    
    QPushButton:hover {
        background-color: #45a049;
    }
    
    QPushButton:pressed {
        background-color: #3d8b40;
    }
)");
```

---

### 2. 选择器

```css
/* 类型选择器 */
QPushButton { background-color: blue; }

/* 类选择器 */
.QPushButton { background-color: blue; }

/* ID 选择器 */
QPushButton#myButton { background-color: red; }

/* 属性选择器 */
QPushButton[flat="true"] { border: none; }

/* 子选择器 */
QDialog > QPushButton { margin: 10px; }

/* 伪状态 */
QPushButton:hover { background-color: green; }
QPushButton:pressed { background-color: darkgreen; }
QPushButton:disabled { color: gray; }
QPushButton:checked { background-color: yellow; }
```

### 6. 样式表注意事项

```cpp
// ❌ 错误：父控件样式覆盖子控件
QWidget* parent = new QWidget();
parent->setStyleSheet("background-color: red;");  // 子控件也会变红

// ✅ 正确：指定类型
parent->setStyleSheet("QWidget { background-color: red; }");
// 或
parent->setStyleSheet(R"(
    QWidget#mainWidget {
        background-color: red;
    }
)");
parent->setObjectName("mainWidget");
```

## 13.4 如何实现国际化（i18n）？

### 答案

Qt 通过 **QTranslator** 和 `tr()` 函数实现国际化。

---

### 1. 标记可翻译字符串

```cpp
class MainWindow : public QMainWindow {
    Q_OBJECT
public:
    MainWindow() {
        setWindowTitle(tr("我的应用"));  // 使用 tr()
        
        QPushButton* button = new QPushButton(tr("确定"), this);
        QLabel* label = new QLabel(tr("欢迎使用"), this);
        
        // 带上下文的翻译
        QString status = tr("Status: %1").arg(tr("Ready"));
        
        // 带复数的翻译
        int count = 5;
        QString msg = tr("%n file(s) selected", "", count);
    }
};

// 在非 QObject 类中使用
QString text = QCoreApplication::translate("Context", "Hello");
```

---

### 2. 生成翻译文件

```qmake
# .pro 文件
TRANSLATIONS += translations/app_zh_CN.ts \
                translations/app_en_US.ts
```

```bash
# 提取可翻译字符串
lupdate myapp.pro

# 生成的 .ts 文件（XML 格式）
```

---

### 3. 翻译 .ts 文件

使用 **Qt Linguist** 工具翻译：

```xml
<!-- app_zh_CN.ts -->
<?xml version="1.0" encoding="utf-8"?>
<!DOCTYPE TS>
<TS version="2.1" language="zh_CN">
<context>
    <name>MainWindow</name>
    <message>
        <source>My Application</source>
        <translation>我的应用</translation>
    </message>
    <message>
        <source>OK</source>
        <translation>确定</translation>
    </message>
    <message>
        <source>Welcome</source>
        <translation>欢迎使用</translation>
    </message>
</context>
</TS>
```

---

### 4. 编译翻译文件

```bash
# 编译 .ts 为 .qm（二进制格式）
lrelease myapp.pro

# 生成 app_zh_CN.qm、app_en_US.qm
```

---

### 5. 加载翻译

```cpp
#include <QTranslator>
#include <QLocale>

int main(int argc, char* argv[]) {
    QApplication app(argc, argv);
    
    // 创建翻译器
    QTranslator translator;
    
    // 根据系统语言加载翻译
    QString language = QLocale::system().name();  // zh_CN, en_US, ...
    QString fileName = QString(":/translations/app_%1.qm").arg(language);
    
    if (translator.load(fileName)) {
        app.installTranslator(&translator);
    }
    
    MainWindow window;
    window.show();
    
    return app.exec();
}
```

---

### 6. 动态切换语言

```cpp
class LanguageSwitcher : public QObject {
    Q_OBJECT
public:
    void switchLanguage(const QString& language) {
        // 移除旧翻译器
        if (m_translator) {
            qApp->removeTranslator(m_translator);
            delete m_translator;
        }
        
        // 加载新翻译器
        m_translator = new QTranslator(this);
        QString fileName = QString(":/translations/app_%1.qm").arg(language);
        
        if (m_translator->load(fileName)) {
            qApp->installTranslator(m_translator);
            
            // 通知所有窗口重新翻译
            emit languageChanged();
        }
    }
    
signals:
    void languageChanged();
    
private:
    QTranslator* m_translator = nullptr;
};

// 窗口响应语言变化
class MainWindow : public QMainWindow {
protected:
    void changeEvent(QEvent* event) override {
        if (event->type() == QEvent::LanguageChange) {
            retranslateUi();  // 重新翻译界面
        }
        QMainWindow::changeEvent(event);
    }
    
    void retranslateUi() {
        setWindowTitle(tr("我的应用"));
        m_button->setText(tr("确定"));
        // ... 更新所有文本
    }
};
```

---

### 7. 翻译最佳实践

```cpp
// ✅ 正确：提供上下文
tr("File")  // 文件（名词）
tr("File", "verb")  // 归档（动词）

// ✅ 正确：使用占位符
tr("User %1 logged in").arg(userName);

// ✅ 正确：复数形式
tr("%n item(s)", "", count);

// ❌ 错误：拼接字符串
tr("User ") + userName + tr(" logged in");  // 不同语言语序可能不同

// ✅ 正确：完整句子
tr("User %1 logged in").arg(userName);
```

## 本章小结

### 核心知识点

1. **插件系统**
   - 定义接口（Q_DECLARE_INTERFACE）
   - 实现插件（Q_PLUGIN_METADATA）
   - 加载插件（QPluginLoader）

2. **资源系统**
   - .qrc 文件定义资源
   - :/ 前缀访问资源
   - 支持压缩和动态加载

3. **样式表（QSS）**
   - 类似 CSS 语法
   - 支持选择器、伪状态
   - 可实现深色主题

4. **国际化（i18n）**
   - tr() 标记可翻译字符串
   - lupdate 提取、lrelease 编译
   - QTranslator 加载翻译

### 最佳实践

- ✅ 插件接口使用纯虚函数
- ✅ 资源文件使用前缀和别名
- ✅ 样式表从文件加载（便于维护）
- ✅ 所有用户可见文本使用 tr()
- ✅ 使用 Qt Linguist 进行翻译
- ✅ 支持动态语言切换
