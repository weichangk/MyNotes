## 10.1 Qt 网络编程有哪些常用类？

### 答案

Qt Network 模块提供了**丰富的网络编程类**，支持 HTTP、TCP、UDP 等协议。

### 核心类对比

| 类 | 协议 | 用途 |
|---|------|------|
| QNetworkAccessManager | HTTP/HTTPS | HTTP 客户端（GET/POST） |
| QNetworkRequest | HTTP | 构建 HTTP 请求 |
| QNetworkReply | HTTP | 接收 HTTP 响应 |
| QTcpSocket | TCP | TCP 客户端 |
| QTcpServer | TCP | TCP 服务器 |
| QUdpSocket | UDP | UDP 通信 |
| QSslSocket | SSL/TLS | 加密通信 |
| QWebSocket | WebSocket | WebSocket 客户端/服务器 |

### 模块依赖

```qmake
# .pro 文件
QT += network
```

## 10.2 如何实现 HTTP 客户端？

### 答案

使用 `QNetworkAccessManager` 发送 HTTP 请求。

### GET 请求示例

```cpp
#include <QNetworkAccessManager>
#include <QNetworkRequest>
#include <QNetworkReply>
#include <QUrl>
#include <QDebug>

class HttpClient : public QObject {
    Q_OBJECT
public:
    HttpClient() {
        m_manager = new QNetworkAccessManager(this);
        connect(m_manager, &QNetworkAccessManager::finished,
                this, &HttpClient::onFinished);
    }
    
    void get(const QString& url) {
        QNetworkRequest request(QUrl(url));
        m_manager->get(request);
    }
    
private slots:
    void onFinished(QNetworkReply* reply) {
        if (reply->error() == QNetworkReply::NoError) {
            // 成功
            QByteArray data = reply->readAll();
            qDebug() << "响应:" << data;
        } else {
            // 失败
            qDebug() << "错误:" << reply->errorString();
        }
        
        reply->deleteLater();
    }
    
private:
    QNetworkAccessManager* m_manager;
};

// 使用
HttpClient client;
client.get("https://api.example.com/data");
```

void getWithParams(const QString& baseUrl, const QMap<QString, QString>& params) {
    QUrl url(baseUrl);
    QUrlQuery query;
    
    for (auto it = params.begin(); it != params.end(); ++it) {
        query.addQueryItem(it.key(), it.value());
    }
    
    url.setQuery(query);
    
    QNetworkRequest request(url);
    m_manager->get(request);
}

// 使用

### 下载文件示例

```cpp
class FileDownloader : public QObject {
    Q_OBJECT
public:
    void download(const QString& url, const QString& savePath) {
        m_savePath = savePath;
        
        QNetworkRequest request(QUrl(url));
        QNetworkReply* reply = m_manager->get(request);
        
        // 监听下载进度
        connect(reply, &QNetworkReply::downloadProgress,
                this, &FileDownloader::onProgress);
        
        connect(reply, &QNetworkReply::finished,
                this, &FileDownloader::onDownloadFinished);
    }
    
private slots:
    void onProgress(qint64 received, qint64 total) {
        if (total > 0) {
            int percent = received * 100 / total;
            qDebug() << "下载进度:" << percent << "%";
        }
    }
    
    void onDownloadFinished() {
        QNetworkReply* reply = qobject_cast<QNetworkReply*>(sender());
        if (!reply) return;
        
        if (reply->error() == QNetworkReply::NoError) {
            QFile file(m_savePath);
            if (file.open(QIODevice::WriteOnly)) {
                file.write(reply->readAll());
                file.close();
                qDebug() << "下载成功:" << m_savePath;
            }
        } else {
            qDebug() << "下载失败:" << reply->errorString();
        }
        
        reply->deleteLater();
    }
    
private:
    QNetworkAccessManager* m_manager = new QNetworkAccessManager(this);
    QString m_savePath;
};

// 使用
FileDownloader downloader;
downloader.download("https://example.com/file.zip", "file.zip");
```

### 上传文件示例

```cpp
void upload(const QString& url, const QString& filePath) {
    QFile* file = new QFile(filePath);
    if (!file->open(QIODevice::ReadOnly)) {
        qDebug() << "无法打开文件";
        delete file;
        return;
    }
    
    QNetworkRequest request(QUrl(url));
    request.setHeader(QNetworkRequest::ContentTypeHeader, "application/octet-stream");
    
    QNetworkReply* reply = m_manager->post(request, file);
    
    // 文件会在 reply 完成后自动删除
    file->setParent(reply);
    
    connect(reply, &QNetworkReply::uploadProgress,
            [](qint64 sent, qint64 total) {
        if (total > 0) {
            qDebug() << "上传进度:" << (sent * 100 / total) << "%";
        }
    });
}
```

## 10.3 如何实现 TCP 服务器和客户端？

### 答案

使用 `QTcpServer` 和 `QTcpSocket` 实现 TCP 通信。

---

### TCP 服务器示例

```cpp
#include <QTcpServer>
#include <QTcpSocket>
#include <QDebug>

class TcpServer : public QObject {
    Q_OBJECT
public:
    TcpServer() {
        m_server = new QTcpServer(this);
        
        connect(m_server, &QTcpServer::newConnection,
                this, &TcpServer::onNewConnection);
    }
    
    bool start(quint16 port) {
        if (!m_server->listen(QHostAddress::Any, port)) {
            qDebug() << "服务器启动失败:" << m_server->errorString();
            return false;
        }
        
        qDebug() << "服务器启动成功，端口:" << port;
        return true;
    }
    
private slots:
    void onNewConnection() {
        QTcpSocket* client = m_server->nextPendingConnection();
        
        qDebug() << "新客户端连接:" << client->peerAddress().toString();
        
        // 读取数据
        connect(client, &QTcpSocket::readyRead, [client]() {
            QByteArray data = client->readAll();
            qDebug() << "收到:" << data;
            
            // 回复数据
            client->write("服务器收到: " + data);
        });
        
        // 断开连接
        connect(client, &QTcpSocket::disconnected, [client]() {
            qDebug() << "客户端断开:" << client->peerAddress().toString();
            client->deleteLater();
        });
        
        m_clients.append(client);
    }
    
public slots:
    // 广播消息
    void broadcast(const QByteArray& data) {
        for (QTcpSocket* client : m_clients) {
            if (client->state() == QAbstractSocket::ConnectedState) {
                client->write(data);
            }
        }
    }
    
private:
    QTcpServer* m_server;
    QList<QTcpSocket*> m_clients;
};

// 使用
TcpServer server;
server.start(8888);
server.broadcast("广播消息");
```

### TCP 客户端示例

```cpp
#include <QTcpSocket>
#include <QDebug>

class TcpClient : public QObject {
    Q_OBJECT
public:
    TcpClient() {
        m_socket = new QTcpSocket(this);
        
        // 连接成功
        connect(m_socket, &QTcpSocket::connected, [this]() {
            qDebug() << "连接成功";
            emit connected();
        });
        
        // 读取数据
        connect(m_socket, &QTcpSocket::readyRead, [this]() {
            QByteArray data = m_socket->readAll();
            qDebug() << "收到:" << data;
            emit dataReceived(data);
        });
        
        // 断开连接
        connect(m_socket, &QTcpSocket::disconnected, [this]() {
            qDebug() << "连接断开";
            emit disconnected();
        });
        
        // 错误处理
        connect(m_socket, &QTcpSocket::errorOccurred,
                [this](QAbstractSocket::SocketError error) {
            qDebug() << "错误:" << m_socket->errorString();
            emit errorOccurred(error);
        });
    }
    
    void connectToHost(const QString& host, quint16 port) {
        qDebug() << "连接到" << host << ":" << port;
        m_socket->connectToHost(host, port);
    }
    
    void send(const QByteArray& data) {
        if (m_socket->state() == QAbstractSocket::ConnectedState) {
            m_socket->write(data);
            m_socket->flush();
        } else {
            qDebug() << "未连接，无法发送";
        }
    }
    
    void disconnect() {
        m_socket->disconnectFromHost();
    }
    
signals:
    void connected();
    void disconnected();
    void dataReceived(const QByteArray& data);
    void errorOccurred(QAbstractSocket::SocketError error);
    
private:
    QTcpSocket* m_socket;
};

// 使用
TcpClient client;
QObject::connect(&client, &TcpClient::connected, []() {
    qDebug() << "已连接，可以发送数据";
});

client.connectToHost("127.0.0.1", 8888);
client.send("Hello Server!");
```

## 10.4 如何实现 UDP 通信？

### 答案

使用 `QUdpSocket` 实现 UDP 通信（无连接、不可靠）。

### UDP 发送端

```cpp
#include <QUdpSocket>
#include <QDebug>

class UdpSender : public QObject {
    Q_OBJECT
public:
    UdpSender() {
        m_socket = new QUdpSocket(this);
    }
    
    void send(const QString& host, quint16 port, const QByteArray& data) {
        qint64 sent = m_socket->writeDatagram(data, QHostAddress(host), port);
        if (sent == -1) {
            qDebug() << "发送失败:" << m_socket->errorString();
        } else {
            qDebug() << "发送成功:" << sent << "字节";
        }
    }
    
    // 广播
    void broadcast(quint16 port, const QByteArray& data) {
        m_socket->writeDatagram(data, QHostAddress::Broadcast, port);
    }
    
private:
    QUdpSocket* m_socket;
};

// 使用
UdpSender sender;
sender.send("192.168.1.100", 9999, "UDP消息");
sender.broadcast(9999, "广播消息");
```

---

### UDP 接收端

```cpp
#include <QUdpSocket>
#include <QDebug>

class UdpReceiver : public QObject {
    Q_OBJECT
public:
    UdpReceiver() {
        m_socket = new QUdpSocket(this);
        
        connect(m_socket, &QUdpSocket::readyRead,
                this, &UdpReceiver::onReadyRead);
    }
    
    bool bind(quint16 port) {
        if (!m_socket->bind(QHostAddress::Any, port)) {
            qDebug() << "绑定失败:" << m_socket->errorString();
            return false;
        }
        
        qDebug() << "监听端口:" << port;
        return true;
    }
    
private slots:
    void onReadyRead() {
        while (m_socket->hasPendingDatagrams()) {
            QByteArray data;
            data.resize(m_socket->pendingDatagramSize());
            
            QHostAddress sender;
            quint16 senderPort;
            
            m_socket->readDatagram(data.data(), data.size(), &sender, &senderPort);
            
            qDebug() << "收到来自" << sender.toString() << ":" << senderPort;
            qDebug() << "数据:" << data;
        }
    }
    
private:
    QUdpSocket* m_socket;
};

// 使用
UdpReceiver receiver;
receiver.bind(9999);
```

### 组播（Multicast）示例

```cpp
class MulticastReceiver : public QObject {
    Q_OBJECT
public:
    bool join(const QString& groupAddress, quint16 port) {
        m_socket = new QUdpSocket(this);
        
        // 绑定端口
        if (!m_socket->bind(QHostAddress::AnyIPv4, port, QUdpSocket::ShareAddress)) {
            qDebug() << "绑定失败";
            return false;
        }
        
        // 加入组播组
        QHostAddress group(groupAddress);
        if (!m_socket->joinMulticastGroup(group)) {
            qDebug() << "加入组播组失败";
            return false;
        }
        
        qDebug() << "加入组播组成功:" << groupAddress;
        
        connect(m_socket, &QUdpSocket::readyRead,
                this, &MulticastReceiver::onReadyRead);
        
        return true;
    }
    
private slots:
    void onReadyRead() {
        while (m_socket->hasPendingDatagrams()) {
            QByteArray data;
            data.resize(m_socket->pendingDatagramSize());
            m_socket->readDatagram(data.data(), data.size());
            qDebug() << "组播消息:" << data;
        }
    }
    
private:
    QUdpSocket* m_socket;
};

// 使用
MulticastReceiver receiver;
receiver.join("239.255.0.1", 9999);
```

## 本章小结

### 核心知识点

1. **HTTP 客户端**
   - QNetworkAccessManager：发送请求
   - QNetworkRequest：构建请求
   - QNetworkReply：接收响应
   - 支持 GET、POST、下载、上传

2. **TCP 通信**
   - QTcpServer：服务器端
   - QTcpSocket：客户端/连接
   - 可靠、有序、基于连接

3. **UDP 通信**
   - QUdpSocket：发送/接收
   - 无连接、不可靠、快速
   - 支持广播和组播

### 协议选择

| 场景 | 推荐协议 |
|------|---------|
| Web API 调用 | HTTP/HTTPS |
| 文件传输 | TCP |
| 实时聊天 | TCP / WebSocket |
| 实时游戏 | UDP |
| 局域网发现 | UDP 广播 |
| 流媒体 | UDP |

### 最佳实践

- ✅ HTTP 请求使用单例 QNetworkAccessManager
- ✅ 始终处理网络错误
- ✅ 使用信号槽处理异步响应
- ✅ QNetworkReply 使用完后调用 deleteLater()
- ✅ TCP 长连接使用心跳保活
- ✅ UDP 包大小不超过 MTU（1500字节）
